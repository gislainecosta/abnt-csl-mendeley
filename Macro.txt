' ============================================================
'  FormataEtAl
'  Aplica itálico em todas as ocorrências de "et al." no
'  documento ativo: corpo, campos do Mendeley, cabeçalhos
'  e rodapés. Ignora ocorrências já em itálico.
' ============================================================

Sub FormataEtAl()

    Dim oDoc     As Document
    Dim oField   As Field
    Dim oSection As Section
    Dim oHF      As HeaderFooter
    Dim contador As Long

    Set oDoc = ActiveDocument
    contador = 0

    ' 1. Corpo principal
    contador = contador + FormatarRange(oDoc.Content)

    ' 2. Campos (citações do Mendeley/Word)
    For Each oField In oDoc.Fields
        contador = contador + FormatarRange(oField.Result)
    Next oField

    ' 3. Cabeçalhos e rodapés
    For Each oSection In oDoc.Sections
        For Each oHF In oSection.Headers
            If oHF.Exists Then contador = contador + FormatarRange(oHF.Range)
        Next oHF
        For Each oHF In oSection.Footers
            If oHF.Exists Then contador = contador + FormatarRange(oHF.Range)
        Next oHF
    Next oSection

    If contador > 0 Then
        MsgBox "Pronto! " & contador & " ocorrência(s) de ""et al."" formatada(s) em itálico.", _
               vbInformation, "FormataEtAl"
    Else
        MsgBox "Nenhuma ocorrência nova de ""et al."" encontrada." & vbCrLf & _
               "(As existentes já podem estar em itálico.)", _
               vbInformation, "FormataEtAl"
    End If

End Sub

' ------------------------------------------------------------
'  FormatarRange: busca "et al." num Range e aplica itálico.
'  Retorna o número de ocorrências formatadas.
' ------------------------------------------------------------
Private Function FormatarRange(oRange As Range) As Long

    Dim contador As Long
    contador = 0

    With oRange.Find
        .ClearFormatting
        .Text = "et al."
        .Forward = True
        .Wrap = wdFindStop
        .Format = False
        .MatchCase = True
        .MatchWholeWord = False
    End With

    Do While oRange.Find.Execute
        If Not oRange.Font.Italic Then
            oRange.Font.Italic = True
            contador = contador + 1
        End If
    Loop

    FormatarRange = contador

End Function
